#perp_perp_para
A B C D E F : perp A B C D, perp C D E F, ncoll A B E => para A B E F

#para_same_line
A B C : para A B A C => coll A B C

#perp_perp_eqangles
A B C D E F G H : perp A B C D, perp E F G H, npara A B E F, npara A B G H => eqangle A B E F C D G H

#para_eqangle_eqangle
A B C D E F G H X Y : eqangle A B C D E F G H, para A B X Y => eqangle X Y C D E F G H

#alt_angles_para
A B C D P Q : eqangle A B P Q C D P Q, ncoll A B C => para A B C D

#cyclic_eqangle
A B P Q : cyclic A B P Q => eqangle P A P B Q A Q B

#eqangle_cyclic
A B P Q : eqangle P A P B Q A Q B => cyclic A B P Q
