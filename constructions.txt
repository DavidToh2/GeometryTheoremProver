##### TRIANGLES

triangle a b c : 
-
-
- a b c = triangle

iso_triangle a b c : 
- 
- eqangle b a b c c b c a, cong a b a c
- a b c = iso_triangle

r_triangle a b c : 
-
- perp a b a c
- a b c = r_triangle

riso_triangle a b c : 
- 
- perp a b a c, cong a b a c, eqangle b a b c c b c a
- a b c = riso_triangle

equi_triangle a b c : 
- diff a b
- cong a b b c, cong b c c a, eqangle a b a c c a c b, eqangle c a c b b c b a
- a b c = equi_triangle

# _p: counterclockwise rotation
r_triangle_p x : a b
- diff a b
- cong x a a b, perp x a a b
- x = r_triangle_p a b

# _n: clockwise rotation
r_triangle_n x : a b
- diff a b
- cong x a a b, perp x a a b
- x = r_triangle_n a b

# _p: counterclockwise rotation of c about b
equi_triangle_p x : b c
- diff b c
- cong x b b c, cong b c c x, eqangle b x b c c b c x, eqangle x c x b b x b c
- x = equi_triangle_p b c

##### ARBITRARY POINT ON OBJECTS

on_line x : a b
- diff a b
- coll x a b
- x = line a b

on_line_at_angle x : a b c d e

on_line_bisect x : a b

- cong x a x b, eqangle a x a b b a b x

on_line_para x : a b c
- diff b c, ncoll a b c
- para x a b c
- x = line_para a b c

on_line_perp x : a b c

- perp x a b c
- x = line_perp a b c

on_ray x : a b

- coll x a b

on_circle x : o a

##### POINT OF INTERSECTIONS OR FIXED POINTS

foot x : a b c
- ncoll a b c
- perp x a b c, coll x b c
- x = line_perp a b c, line b c

midpoint x : a b
- diff a b
- midp x a b
- x = midpoint a b

# a, x, y, b are in this order
trisegment x y : a b
- diff a b
- coll x a b, coll y a b, cong x a x y, cong y x y b
- x y = trisegment a b

mirror x : a b
- diff a b
- coll x a b, cong b a b x
- x = mirror a b

reflect x : a b c
- diff b c, ncoll a b c
- cong b a b x, cong c a c x, perp b c a x
- x = reflect a b c

intersection_cc x : o w a
- ncoll o w a
- cong o a o x, cong w a w x
- x = circle o o a, circle w w a

intersection_lc x : a o b
- diff a b, diff o b, nperp b o b a
- coll x a b, cong o b o x
- x = line a b, circle o o b

intersection_ll x : a b c d
- npara a b c d, ncoll a b c d
- coll x a b, coll x c d
- x = line a b, line c d

##### TRIANGLE CENTERS

circumcenter x : a b c
- ncoll a b c
- cong x a x b, cong x b x c
- x = line_bisect a b, line_bisect a c

incenter i : a b c
- ncoll a b c
- eqangle a b a i a i a c, eqangle c a c i c i c b, eqangle b c b i b i b a
- i = angle_bisect a b c, angle_bisect b c a

incenter2 x y z i : a b c
- ncoll a b c
- eqangle a b a i a i a c, eqangle c a c i c i c b, eqangle b c b i b i b a, coll x b c, perp i x b c, coll y c a, perp i y c a, coll z a b, perp i z a b, cong i x i y, cong i y i z
- i = angle_bisect a b c, angle_bisect b c a; x = line_perp i b c, line b c; y = line_perp i c a, line c a; z = line_perp i a b, line a b

excenter i : a b c
- ncoll a b c
- eqangle a b a i a i a c, eqangle c a c i c i c b, eqangle b c b i b i b a
- i = angle_bisect b a c, angle_exbisect b c a

excenter2 x y z i : a b c
- ncoll a b c
- eqangle a b a i a i a c, eqangle c a c i c i c b, eqangle b c b i b i b a, coll x b c, perp i x b c, coll y c a, perp i y c a, coll z a b, perp i z a b, cong i x i y, cong i y i z
- i = angle_bisect b a c, angle_exbisect b c a; x = line_perp i b c, line b c; y = line_perp i c a, line c a; z = line_perp i a b, line a b

centroid x y z g : a b c
- ncoll a b c
- coll x b c, cong x b x c, coll y c a, cong y c y a, coll z a b, cong z a z b, coll a x g, coll b y g, coll c z g
- x = midpoint b c; y = midpoint c a; z = midpoint a b; g = line a x, line b y

orthocenter h : a b c
- ncoll a b c
- perp h a b c, perp h b c a, perp h c a b
- h = line_perp a b c, line_perp b c a, line_perp c a b

orthocenter2 x y z h : a b c
- ncoll a b c
- perp a x b c, perp b y a c, perp c z a b, coll a h x, coll b h y, coll c h z, coll b x c, coll c y a, coll a z b
- x = line_perp a b c, line b c; y = line_perp b a c, line a c; z = line_perp c a b, line a b; h = line a x, line b y, line c z

##### TANGENTS

# Tangent to circle centered at o at point a
lc_tangent x : a o
- diff a o
- perp a x a o
- x = line_perp a a o

# Tangents from point a to circle centered at o with radius ob
tangents x y : a o b
- diff o a, diff o b, diff a b
- cong o x o b, perp a x o x, cong o y o b, perp a y o y
- x y = tangents a o b

# Single common EXTERNAL tangent
common_tangent x y : o a i b
- diff o a, diff i b, diff o i
- cong o x o a, cong i y i b, perp x o x y, perp y i y x
- x y = common_tangent o a i b

# A pair of common EXTERNAL tangents
common_tangent2 x y z w : o a i b
- diff o a, diff i b, diff o i
- cong o x o a, cong i y i b, perp x o x y, perp y i y x, cong o z o a, cong i w i b, perp z o z w, perp w i w z
- x y z w = common_tangent2 o a i b
