problem p1a {
    a d e = triangle a d e;
    b = on_line b : a d;
    c = on_line_para c : b d e, on_line c : a e
    ? simtri a b c a d e
}

problem p1b {
    a d e = triangle a d e;
    b = on_line b : a d;
    c = on_line_at_angle c : b a e d a, on_line c : a e
    ? simtri a b c a d e
}
